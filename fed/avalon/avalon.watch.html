    $watch方法是位于VM中，当VM的第一层的属性发生改变时，就会执行预先绑定的回调。

    一个简单的例子

               var model = avalon.define("test1", function(vm) {
                vm.prop = 111
                vm.$watch("prop", function(newValue,oldValue){
                    console.log(newValue)
                    console.log(oldValue)
                    console.log("---------------")
                })

            });
            setTimeout(function(){
                model.prop = 222
            }, 1000)

    打印出：

                222
            111
            ---------------

    用于在$watch方法进行数据校验非常方便。

    例子2， 对计算属性依赖的某个监控属性进行赋值

                var model = avalon.define("test", function(vm) {
                vm.firstName = "司徒";
                vm.lastName = "正美"
                vm.fullName = {
                    set: function(val) {
                        var array = val.split(" ")
                        this.firstName = array[0] 
                        this.lastName = array[1] 
                    },
                    get: function() {
                        return this.firstName + " " + this.lastName;
                    }
                };
                vm.$watch("fullName", function(newValue, oldValue) {
                    console.log(newValue)
                    console.log(oldValue)
                    console.log("---------------")
                })

            });
            setTimeout(function() {
                model.firstName = "欧阳"
            }, 1000)
            
                欧阳 正美
            司徒 正美
            ---------------

    例子3， 对计算属性重新赋值

                var model = avalon.define("test", function(vm) {
                vm.firstName = "司徒";

                vm.fullName = {
                    set: function(val) {

                        this.lastName = array[1]
                    },
                    get: function() {

                avalon.define("test", function(vm) {
                vm.test0 = false;
                vm.test1 = {
                    set: function(val) {
                        this.test0 = val;
                    },
                    get: function() {
                        return this.test0;
                    }
                };
                vm.test2 = false;
                vm.$watch('test1', function(val) {
                    vm.test2 = val;
                });
                vm.one = function() {
                    vm.test1 = !vm.test1;
                };
                vm.two = function() {
                    vm.test0 = !vm.test0;
                };
            });

    内部运行流程

    当我们点击测试1时，test1由false变成true, 并通过$watch带动test2也变成true。

    当我们点击测试2时，test0由false变成true, 由于test1与test0存在依赖关系，test1的值由test0计算得到。在最开始的时候，test1与test0都是false,而现在test1也与test0变成true了，因此也通过$watch带动test2也变成true。

                        }
                        vm.msg += '！！';
                    }
                });
                vm.$watch('test1', function(val) {
                    if (val) {
                        vm.msg += 'test1-';
                        vm.test2 = true;
                    }
                });
                vm.one = function() {
                    vm.test0 = true;
                };
            });
