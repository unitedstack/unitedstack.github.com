    参照以下的结构，引用JS与CSS，一行代码不用写就有效果了。

    它基于jQuery ui的slider组件改成，直接由800行（还没有计它其他依赖文件）减少成200行。

    我们看下面的例子：

            <div style="width:70%">
            <input ms-ui="slider"data-id="aaa"  data-max="255" data-value="36">
        </div>
        <p ms-controller="aaa">{{value}}  {{percent}}</p>
        <div style="width:70%">
            <input ms-ui="slider" data-range="min">
        </div>
        <br/>
        <div style="width:70%">
            <p>Range with fixed maximum</p>
            <input ms-ui="slider" data-range="max"  data-max="10" data-value="2">
            <p>Range with fixed minimum</p>
            <input  ms-ui="slider" data-min="1" data-max="700" data-value="37" data-range="min"/>
            <p>Snap to increments</p>
            <input  ms-ui="slider" data-min="0" data-max="500" data-value="100" data-step="50" />

        </div>

        <div style="height:200px">
            <input  ms-ui="slider"  data-value="20" data-range="min" data-orientation="vertical"  data-id="bbb">
            <input  ms-ui="slider"  data-value="40" data-range="min" data-orientation="vertical" >
            <input  ms-ui="slider"  data-value="60" data-range="min" data-orientation="vertical" >
            <input  ms-ui="slider"  data-value="80" data-range="min" data-orientation="vertical" >
            <input  ms-ui="slider"  data-value="80" data-range="true" data-orientation="vertical" >
            <!--            <input  ms-ui="slider"  data-value="60" data-range="max" data-orientation="vertical" >-->
        </div>  
        <p ms-controller="bbb">{{ value }}</p>
        <div style="width:70%">
            <input ms-ui="slider" data-id="ccc" data-range="true" data-values="[ 25,50 ]" data-max="100">
            <p ms-controller="ccc">{{ percent0 }}  {{ percent1 }}</p>
            <p ms-controller="ccc">{{ value }}</p>
        </div>
