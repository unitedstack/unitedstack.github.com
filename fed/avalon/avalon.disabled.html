    • ms-disabled, ms-enabled, ms-readonly, ms-checked, ms-selected绑定

    返回首页

    这五个绑定是用于表单元素，控制它们的disabled, readonly, checked, selected这四个布尔属性，其中ms-enabled是ms-disabled的反转操作，目的为了节约脑细胞。

    用法ms-disabled="aaa"，aaa可以是ViewModel中的某个监控属性或表达式，如aaa+1, 4 > 1什么的

                <div ms-controller="props">
                <div><input ms-disabled="toggle"></div>
                <div><input ms-enabled="toggle"></div>
                <div><input ms-readonly="toggle"></div>
                <div><input type="radio" ms-model="toggle"></div>
                <div><input type="checkbox" ms-checked="toggle"></div>
                <div><input ms-model="selectedIndex"></div>
                <div>
                    <select ms-each-el="array">
                        <option ms-selected="$index == selectedIndex" >{{el}}</option>
                    </select>
                </div>
            </div>
                
             avalon.ready(function(){
            avalon.define("props", function(vm) {
                vm.toggle = true
                vm.array = ["苹果", "香蕉", "桃子", "雪梨", "葡萄", "哈蜜瓜","橙子","火龙果","荔技","黄皮"]
                vm.selectedIndex = 2;
            });
           avalon.scan();
         })
